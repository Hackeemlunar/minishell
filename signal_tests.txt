# This file contains instructions for testing signal handling in minishell

echo "=== Signal Handling Test Instructions ==="
echo "Run these tests in both your minishell and bash to compare behavior."
echo ""

echo "Test 1: Ctrl+C while in prompt"
echo "------------------------------"
echo "1. Start minishell"
echo "2. Wait at the prompt"
echo "3. Press Ctrl+C"
echo "4. Expected behavior: Display a new prompt on a new line"
echo ""

echo "Test 2: Ctrl+C while running a command that doesn't terminate"
echo "------------------------------------------------------------"
echo "1. Run: cat"
echo "2. Press Ctrl+C"
echo "3. Expected behavior: Command terminates and returns to prompt"
echo ""

echo "Test 3: Ctrl+C during heredoc"
echo "----------------------------"
echo "1. Start a heredoc: cat << EOF"
echo "2. Type a few lines"
echo "3. Press Ctrl+C before typing EOF"
echo "4. Expected behavior: Heredoc terminates and returns to prompt (possibly with error)"
echo ""

echo "Test 4: Ctrl+D at empty prompt"
echo "----------------------------"
echo "1. Start minishell"
echo "2. At an empty prompt, press Ctrl+D"
echo "3. Expected behavior: Shell exits gracefully"
echo ""

echo "Test 5: Ctrl+D with text at prompt"
echo "--------------------------------"
echo "1. Start minishell"
echo "2. Type some text but don't press Enter"
echo "3. Press Ctrl+D"
echo "4. Expected behavior: Nothing happens (in bash) or executes the line (in some shells)"
echo ""

echo "Test 6: Ctrl+D during command input"
echo "--------------------------------"
echo "1. Run: cat"
echo "2. Type some text and press Enter"
echo "3. Press Ctrl+D on an empty line"
echo "4. Expected behavior: cat terminates and returns to prompt"
echo ""

echo "Test 7: Ctrl+C during a pipe sequence"
echo "-----------------------------------"
echo "1. Run: cat | grep a"
echo "2. Type some text"
echo "3. Press Ctrl+C"
echo "4. Expected behavior: Both processes terminate and return to prompt"
echo ""

echo "Test 8: Multiple Ctrl+C presses"
echo "-----------------------------"
echo "1. Start minishell"
echo "2. Press Ctrl+C multiple times in quick succession"
echo "3. Expected behavior: Display a new prompt for each Ctrl+C"
echo ""

echo "Test 9: Ctrl+\ (SIGQUIT) handling"
echo "-------------------------------"
echo "1. Run: cat"
echo "2. Press Ctrl+\\"
echo "3. Expected behavior: Should be ignored in interactive mode"
echo ""

echo "=== Notes on Signal Handling ==="
echo "- SIGINT (Ctrl+C) should interrupt the current foreground process"
echo "- EOF (Ctrl+D) should be handled as an exit command when at an empty prompt"
echo "- SIGQUIT (Ctrl+\\) should be ignored in interactive mode"
echo "- The terminal settings should be restored after signal handling"
